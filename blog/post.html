(async () => { const slugFromURL = decodeURIComponent( window.location.pathname.replace(/^\/blog\//, '').replace(/\/+$/, '').trim() ); if (!slugFromURL) { showError(); return; } try { const res = await fetch(`/_posts-json/${slugFromURL}.json`); if (!res.ok) throw new Error(`HTTP ${res.status}`); const data = await res.json(); const title = data.title || 'مقال على NEORA'; document.title = `${title} — NEORA`; document.getElementById('canonical').href = window.location.href; setMeta('meta-desc', data.description || ''); setMeta('og-title', title); setMeta('og-desc', data.description || ''); document.getElementById('post-title').textContent = title; if (data.date) { const d = new Date(data.date); document.getElementById('post-meta').innerHTML = `📅 ${d.toLocaleDateString('ar-SA', {year:'numeric', month:'long', day:'numeric'})}`; } if (data.image) { const img = document.getElementById('post-image'); img.src = data.image; img.alt = title; img.style.display = 'block'; } marked.setOptions({ breaks: true, gfm: true }); document.getElementById('post-body').innerHTML = marked.parse(data.content || ''); document.getElementById('loading').style.display = 'none'; document.getElementById('post-content').style.display = 'block'; } catch (e) { console.error('Error:', e); showError(); } })(); function showError() { document.getElementById('loading').style.display = 'none'; document.getElementById('error-box').style.display = 'block'; } function setMeta(id, val) { if (!val) return; const el = document.getElementById(id); if (el) el.setAttribute('content', val); }
